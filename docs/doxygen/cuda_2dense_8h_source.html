52 inline tvm::te::Tensor dense_cuda(const Target& target,
53  const tvm::te::Tensor& data,
54  const tvm::te::Tensor& weight,
55  const tvm::te::Tensor& bias,
56  const DataType& out_dtype) {
57  CHECK_EQ(data->shape.size(), 2) << "dense requires 2-D data";
58  CHECK_EQ(weight->shape.size(), 2) << "dense requires 2-D weight";
59  if (bias.defined()) {
60  CHECK_EQ(bias->shape.size(), 1) << "dense requires 1-D bias";
61  }
62 
63  auto batch = data->shape[0];
64  auto in_dim = data->shape[1];
65  auto out_dim = weight->shape[0];
66 
67  if (target->libs().count("cublas")) {
68  CHECK_EQ(data->dtype, out_dtype) << "Mixed precision not supported.";
69  auto mm = topi::contrib::cublas_matmul(data, weight, false, true);
70  if (bias.defined()) {
71  mm = tvm::te::compute({ batch, out_dim },
72  [&](Var i, Var j) {
73  return mm(i, j) + bias(j);
74  }, "tensor", kBroadcast);
75  }
76 
77  return mm;
78  } else {
79  return topi::nn::dense(data, weight, bias, out_dtype);
80  }

91 inline Schedule schedule_dense(const Target &target, const Array<Tensor>& outs) {
92  if (target->target_name == "cuda" &&
93  target->libs().count("cublas")) {
94  return topi::generic::schedule_extern(target, outs);
95  }
96 
97  Array<Operation> out_ops;
98  for (auto t : outs) {
99  out_ops.push_back(t->op);
100  }
101  auto s = create_schedule(out_ops);
102 
103  auto _schedule = [&](const Tensor& dense) {
104  auto num_thread = 64;
105  auto k = dense->op.as<ComputeOpNode>()->reduce_axis[0];
106  IterVar ko, kf;
107  s[dense].split(k, num_thread, &ko, &kf);
108  auto dense_f = s.rfactor(dense, kf)[0];
109 
110  Tensor out;
111  if (detail::contains(s->outputs, dense->op)) {
112  out = dense;
113  } else {
114  out = outs[0]->op.output(0);
115  s[dense].compute_at(s[out], s[out]->op.as<ComputeOpNode>()->axis[1]);
116  }
117  s[out].bind(s[out]->op.as<ComputeOpNode>()->axis[0],
118  tvm::te::thread_axis(Range(), "blockIdx.y"));
119  s[out].bind(s[out]->op.as<ComputeOpNode>()->axis[1],
120  tvm::te::thread_axis(Range(), "blockIdx.x"));
121 
122  auto tx = s[dense]->op.as<ComputeOpNode>()->reduce_axis[0];
123  auto thread_x = tvm::te::thread_axis(Range(), "threadIdx.x");
124  s[dense].bind(tx, thread_x);
125  s[dense_f].compute_at(s[dense], tx);
126  s[dense].set_store_predicate(static_cast<PrimExpr>(thread_x) == 0);
127  s[out].set_store_predicate(static_cast<PrimExpr>(thread_x) == 0);
128  };
129 
130  std::function<void(Operation)> traverse;
131  traverse = [&](const Operation& op) {
132  // Inline all one-to-one-mapping operators except the last stage (output)
133  if (is_broadcast(op->tag)) {
134  if (!detail::contains(s->outputs, op)) {
135  s[op].compute_inline();
136  }
137  for (auto tensor : op->InputTensors()) {
138  if (tensor->op->InputTensors().size() > 0) {
139  traverse(tensor->op);
140  }
141  }
142  } else if (op->tag == "dense") {
143  // If tag starts with global_pool
144  auto dense = op.output(0);
145  _schedule(dense);
146  } else {
147  LOG(ERROR) << "Unsupported operator " << op->tag;
148  }
149  };
150 
151  traverse(outs[0]->op);
152  return s;
153 }
