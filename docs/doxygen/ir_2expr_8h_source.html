454 inline TVMPODValue_::operator tvm::PrimExpr() const {
455  if (type_code_ == kTVMNullptr) return PrimExpr();
456  if (type_code_ == kDLInt) {
457  CHECK_LE(value_.v_int64, std::numeric_limits<int>::max());
458  CHECK_GE(value_.v_int64, std::numeric_limits<int>::min());
459  return PrimExpr(static_cast<int>(value_.v_int64));
460  }
461  if (type_code_ == kDLFloat) {
462  return PrimExpr(static_cast<float>(value_.v_float64));
463  }
464 
465  TVM_CHECK_TYPE_CODE(type_code_, kTVMObjectHandle);
466  Object* ptr = static_cast<Object*>(value_.v_handle);
467  return PrimExpr::FromObject_(ObjectPtr<Object>(ptr));
468 }
