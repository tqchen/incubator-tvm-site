727 template<typename ValueType>
728 inline PrimExpr MakeConstScalar(DataType t, ValueType value) {
729  if (t.is_int()) return IntImm(t, static_cast<int64_t>(value));
730  if (t.is_uint()) {
731  // Use IntImm if it is a small integer
732  uint64_t uval = static_cast<uint64_t>(value);
733  if (uval <= static_cast<uint64_t>(std::numeric_limits<int64_t>::max())) {
734  return IntImm(t, static_cast<int64_t>(value));
735  } else {
736  uint64_t mask = (static_cast<uint64_t>(1) << 32U) - 1U;
737  uint64_t low = uval & mask;
738  uint64_t high = uval >> 32U;
739  return LargeUIntImm(t, static_cast<int64_t>(low), static_cast<int64_t>(high));
740  }
741  }
742  if (t.is_float()) return FloatImm(t, static_cast<double>(value));
743  // For now, we store const scalar values of custom datatypes within doubles; later, during the
744  // datatypes lowering pass, we will lower the value to its true representation in the format
745  // specified by the datatype.
746  // TODO(gus) when do we need to start worrying about doubles not being precise enough?
747  if (static_cast<uint8_t>(t.code()) >= static_cast<uint8_t>(kTVMCustomBegin)) {
748  return FloatImm(t, static_cast<double>(value));
749  }
750  LOG(FATAL) << "cannot make const for type " << t;
751  return PrimExpr();
752 }
