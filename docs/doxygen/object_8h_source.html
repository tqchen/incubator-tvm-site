811 template<typename TargetType>
812 inline bool Object::IsInstance() const {
813  const Object* self = this;
814  // NOTE: the following code can be optimized by
815  // compiler dead-code elimination for already known constants.
816  if (self != nullptr) {
817  // Everything is a subclass of object.
818  if (std::is_same<TargetType, Object>::value) return true;
819  if (TargetType::_type_final) {
820  // if the target type is a final type
821  // then we only need to check the equivalence.
822  return self->type_index_ == TargetType::RuntimeTypeIndex();
823  } else {
824  // if target type is a non-leaf type
825  // Check if type index falls into the range of reserved slots.
826  uint32_t begin = TargetType::RuntimeTypeIndex();
827  // The condition will be optimized by constant-folding.
828  if (TargetType::_type_child_slots != 0) {
829  uint32_t end = begin + TargetType::_type_child_slots;
830  if (self->type_index_ >= begin && self->type_index_ < end) return true;
831  } else {
832  if (self->type_index_ == begin) return true;
833  }
834  if (!TargetType::_type_child_slots_can_overflow) return false;
835  // Invariance: parent index is always smaller than the child.
836  if (self->type_index_ < TargetType::RuntimeTypeIndex()) return false;
837  // The rare slower-path, check type hierachy.
838  return self->DerivedFrom(TargetType::RuntimeTypeIndex());
839  }
840  } else {
841  return false;
842  }
843 }
