519  result.push_back(

530  }
531

533  }, name, tag));

552 inline Tensor strided_slice(const Tensor& x,
553  const Array<Integer>& begin,
554  const Array<Integer>& end,
555  const Array<Integer>& strides,
556  std::string name = "T_strided_slice",
557  std::string tag = kInjective) {
558  size_t src_tensor_dim = static_cast<size_t>(x->shape.size());
559  // Setup the ranges.
560  // NOTE: this code duplicates the shape inference logic relay.op
561  // Consider to refactor in the future.
562  std::vector<int64_t> stride_vec;
563  for (Integer i : strides) {
564  CHECK(i.defined());
565  stride_vec.push_back(i->value);
566  }
567  for (size_t i = stride_vec.size(); i < src_tensor_dim; ++i) {
568  stride_vec.push_back(1);
569  }
570  const int64_t max_range = std::numeric_limits<int64_t>::max();
571 
572  std::vector<int64_t> begin_vec;
573  for (size_t i = 0; i < begin.size(); ++i) {
574  if (!begin[i].defined()) {
575  // value=None
576  begin_vec.push_back(stride_vec[i] > 0 ? 0 : max_range);
577  } else {
578  begin_vec.push_back(begin[i]->value);
579  }
580  }
581  for (size_t i = begin_vec.size(); i < src_tensor_dim; ++i) {
582  begin_vec.push_back(stride_vec[i] > 0 ? 0 : max_range);
583  }
584 
585  std::vector<int64_t> end_vec;
586  for (size_t i = 0; i < end.size(); ++i) {
587  // allow end to be None
588  if (!end[i].defined()) {
589  end_vec.push_back(stride_vec[i] < 0 ? 0 : max_range);
590  } else {
591  end_vec.push_back(end[i]->value);
592  }
593  }
594  for (size_t i = end_vec.size(); i < src_tensor_dim; ++i) {
595  end_vec.push_back(stride_vec[i] < 0 ? 0 : max_range);
596  }
597  // Compute
598  Array<PrimExpr> out_shape;
599  Array<PrimExpr> begin_expr;
600  Array<PrimExpr> strides_expr;
601 
602  for (size_t i = 0; i < src_tensor_dim; ++i) {
603  int64_t begin_range = stride_vec[i] < 0 ? -1 : 0;
604  int64_t dim_i = GetConstInt(x->shape[i]);
605  int64_t end_range = stride_vec[i] < 0 ? dim_i - 1 : dim_i;
606  // transform negative indices to positive value, clips on the correct range
607  auto index_canonicalization = [dim_i, begin_range, end_range](int64_t index) {
608  if (index < 0) {
609  index += dim_i;
610  }
611  return std::min(std::max(index, begin_range), end_range);
612  };
613 
614  int64_t begin_i = index_canonicalization(begin_vec[i]);
615  int64_t end_i = index_canonicalization(end_vec[i]);
616 
617  int interval = std::abs(end_i - begin_i);
618  int slice_size = static_cast<int>((interval
619  + std::abs(stride_vec[i]) - 1) / std::abs(stride_vec[i]));
620  CHECK(stride_vec[i] < 0 ? (end_i <= begin_i) : (begin_i <= end_i))
621  << ": Input [Begin=" << begin_vec[i] << ", End=" << end_vec[i]
622  << "] is invalid for axis=" << i;
623 
624  begin_expr.push_back(make_const(begin[0].dtype(), begin_i));
625  strides_expr.push_back(make_const((strides.size() != 0 ? strides[0].dtype() : begin[0].dtype()),
626  stride_vec[i]));
627  out_shape.push_back(slice_size);
628  }
629 
630  return compute(
631  out_shape, [&](const Array<Var>& indices) {
632  Array<PrimExpr> real_indices;
633  for (size_t i = 0; i < src_tensor_dim; ++i) {
634  real_indices.push_back(indices[i] * strides_expr[i] + begin_expr[i]);
635  }
636  return x(real_indices);
637  }, name, tag);
638 }

780  int axis,

782  std::string name = "T_take",
783  std::string tag = kInjective) {

785  axis += static_cast<int>(a->shape.size());

795  for (size_t j = 0; j < indices->shape.size(); ++j) {

811  real_indices.push_back(out_index[j]);

814  axis_dim - 1);
815  real_indices.push_back(idx);
816  for (size_t j = axis + indices_len; j < out_index.size(); ++j) {
817  real_indices.push_back(out_index[j]);
818  }
819  return a(real_indices);
820  }, name, tag);
821  } else if (mode == "fast") {

823  "Make sure input indices are in bound";

825  out_shape, [&](const Array<Var>& out_index) {
826  Array<PrimExpr> indices_position;
827  for (size_t j = axis; j < static_cast<size_t>(axis+indices_len); ++j) {
828  indices_position.push_back(out_index[j]);
829  }
830  Array<PrimExpr> real_indices;
831  for (size_t j = 0; j < static_cast<size_t>(axis); ++j) {
832  real_indices.push_back(out_index[j]);
833  }
834  real_indices.push_back(indices(indices_position));
835  for (size_t j = axis + indices_len; j < out_index.size(); ++j) {
836  real_indices.push_back(out_index[j]);
837  }
838  return a(real_indices);
839  }, name, tag);
840  } else { // mode == "wrap"
841  return compute(
842  out_shape, [&](const Array<Var>& out_index) {
843  Array<PrimExpr> indices_position;
844  for (size_t j = axis; j < static_cast<size_t>(axis+indices_len); ++j) {
845  indices_position.push_back(out_index[j]);
846  }
847  Array<PrimExpr> real_indices;
848  for (size_t j = 0; j < static_cast<size_t>(axis); ++j) {
849  real_indices.push_back(out_index[j]);
850  }
851  auto idx = truncmod(truncmod(indices(indices_position), axis_dim) + axis_dim, axis_dim);
852  real_indices.push_back(idx);

854  real_indices.push_back(out_index[j]);
855  }
856  return a(real_indices);

970 inline Tensor tile(const Tensor& x,
971  Array<Integer> reps,
972  std::string name = "T_tile",
973  std::string tag = kBroadcast) {

975  size_t rdim = reps.size();
976  size_t tdim = (ndim > rdim) ? ndim : rdim;
977  Array<PrimExpr> data_shape;
978  Array<PrimExpr> reps_shape;
979  Array<PrimExpr> new_shape;
980  if (ndim == rdim) {
981  for (size_t i = 0; i < ndim; ++i) {
982  data_shape.push_back(x->shape[i]);
983  reps_shape.push_back(reps[i]);
984  }
985  } else if (ndim > rdim) {
986  for (size_t i = 0; i < ndim; ++i)
987  data_shape.push_back(x->shape[i]);
988  for (size_t i = 0; i < (ndim - rdim); ++i)
989  reps_shape.push_back(1);
990  for (size_t i = 0; i < rdim; ++i)
991  reps_shape.push_back(reps[i]);
992  } else {
993  for (size_t i = 0; i < (rdim - ndim); ++i)
994  data_shape.push_back(1);
995  for (size_t i = 0; i < ndim; ++i)
996  data_shape.push_back(x->shape[i]);
997  for (size_t i = 0; i < rdim; ++i)
998  reps_shape.push_back(reps[i]);
999  }
1000  for (size_t i = 0; i < tdim; ++i)
1001  new_shape.push_back(data_shape[i] * reps_shape[i]);
1002 
1003  if (is_empty_shape(new_shape)) {
1004  return compute(new_shape,
1005  [&](const Array<Var>& indices) { return tvm::cast(x->dtype, 0);},
1006  name, tag);
1007  } else {
1008  return compute(
1009  new_shape, [&](const Array<Var>& indices) {
1010  Array<PrimExpr> idx;
1011  if (ndim >= rdim) {
1012  for (size_t i = 0; i < ndim; ++i)
1013  idx.push_back(indexmod(indices[i], x->shape[i]));
1014  } else {
1015  for (size_t i = 0; i < ndim; ++i)
1016  idx.push_back(indexmod(indices[rdim - ndim + i], x->shape[i]));
1017  }
1018  return x(idx);
1019  }, name, tag);
