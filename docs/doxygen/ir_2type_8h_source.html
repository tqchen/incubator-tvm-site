414 class FuncTypeNode : public TypeNode {
415  public:
417  Array<Type> arg_types;
419  Type ret_type;
420  // The following fields are used in polymorphic(template) functions
421  // For normal functions, the following two fields will be empty.
423  Array<TypeVar> type_params;
428  Array<TypeConstraint> type_constraints;
429 
430  void VisitAttrs(AttrVisitor* v) {
431  v->Visit("arg_types", &arg_types);
432  v->Visit("ret_type", &ret_type);
433  v->Visit("type_params", &type_params);
434  v->Visit("type_constraints", &type_constraints);
435  v->Visit("span", &span);
436  }
437 
438  bool SEqualReduce(const FuncTypeNode* other, SEqualReducer equal) const {
439  // type params first as they defines type vars.
440  return
441  equal.DefEqual(type_params, other->type_params) &&
442  equal(arg_types, other->arg_types) &&
443  equal(ret_type, other->ret_type) &&
444  equal(type_constraints, other->type_constraints);
445  }
446 
447  void SHashReduce(SHashReducer hash_reduce) const {
448  hash_reduce.DefHash(type_params);
449  hash_reduce(arg_types);
450  hash_reduce(ret_type);
451  hash_reduce(type_constraints);
452  }
453

455  TVM_DECLARE_FINAL_OBJECT_INFO(FuncTypeNode, TypeNode);
