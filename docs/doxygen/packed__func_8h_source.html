1063  t.code = kTVMOpaqueHandle;

1065  scan = s.c_str() + 6;
1066  } else if (s == "bool") {
1067  t.code = kDLUInt;
1068  t.bits = 1;
1069  t.lanes = 1;
1070  return t;
1071  } else if (s.substr(0, 6) == "custom") {
1072  t.code = ParseCustomDatatype(s, &scan);
1073  } else {
1074  scan = s.c_str();

1078  uint8_t bits = static_cast<uint8_t>(strtoul(scan, &xdelim, 10));
