49 inline Tensor softmax(const Tensor &x,
50  int axis = -1,
51  std::string name = "tensor",
52  std::string tag = "softmax_output") {
53  auto input_shape = x->shape;
54  auto ndim = input_shape.size();
55  if (axis < 0) {
56  axis = ndim + axis;
57  }
58  CHECK_LT(axis, ndim) << "axis parameter should be less than input dim";
59 
60  auto k1 = tvm::te::reduce_axis(Range(0, input_shape[axis]), "k1");
61  auto k2 = tvm::te::reduce_axis(Range(0, input_shape[axis]), "k2");
62  auto reduced_shape = MakeReduceTargetShape({axis}, x, false, false);
63 
64  tvm::Map<std::string, ObjectRef> attrs;
65  attrs.Set("axis", Integer(axis));
66 
67  auto insert_reduce_index = [axis, ndim](const Array<Var> &indices,
68  const IterVar &reduce_index) {
69  Array<PrimExpr> eval_range;
70  int arg_counter = 0;
71  for (size_t i = 0; i < ndim; ++i) {
72  if (static_cast<int>(i) == axis)
73  eval_range.push_back(reduce_index);
74  else
75  eval_range.push_back(indices[arg_counter++]);
76  }
77  return eval_range;
78  };

80  auto get_non_reduce_indices = [axis, ndim](const Array<Var> &indices) {
81  Array<PrimExpr> non_reduce_indices;
82  for (size_t i = 0; i < ndim; ++i) {
83  if (static_cast<int>(i) != axis)
84  non_reduce_indices.push_back(indices[i]);
85  }
86  return non_reduce_indices;
87  };

89  auto _compute_max = [&](const Array<Var> &indices) {
90  auto eval_range = insert_reduce_index(indices, k1);
91  return topi::MaxOp(x(eval_range), {k1});
92  };

94  auto _compute_exp = [&](const Tensor &max_elem,
95  const Array<Var> &indices) {
96  auto non_reduce_indices = get_non_reduce_indices(indices);
97  return tvm::exp(x(indices) - max_elem(non_reduce_indices));
98  };

100  auto _compute_expsum = [&](const Tensor &exp,
101  const Array<Var> &indices) {
102  auto eval_range = insert_reduce_index(indices, k2);
103  return tvm::sum(exp(eval_range), {k2});
104  };

106  auto _normalize = [&](const Tensor &exp, const Tensor &expsum,
107  const Array<Var> &indices) {
108  auto non_reduce_indices = get_non_reduce_indices(indices);
109  return exp(indices) / expsum(non_reduce_indices);
110  };

112  auto max_elem = tvm::te::compute(reduced_shape, _compute_max);
113  auto exp = tvm::te::compute(input_shape, [&](const Array<Var> &indices) {
114  return _compute_exp(max_elem, indices);
115  });
116  auto expsum = tvm::te::compute(reduced_shape, [&](const Array<Var> &indices) {
117  return _compute_expsum(exp, indices);
118  });
119  return tvm::te::compute(input_shape, [&](const Array<Var> &indices) {
120  return _normalize(exp, expsum, indices);
121  }, name, tag, attrs);

133 inline Tensor log_softmax(const Tensor& x,
134  std::string name = "tensor",
135  std::string tag = "log_softmax_output") {
136  CHECK_EQ(x->shape.size(), 2) << "Log softmax requires 2-D input";
137 
138  PrimExpr m = x->shape[0];
139  PrimExpr n = x->shape[1];
140 
141  auto k = tvm::te::reduce_axis(Range(0, n), "k");
142  auto max_elem = tvm::te::compute(
143  { m }, [&](Var i) {
144  return tvm::max(x(i, k), Array<IterVar>{ k }); });
145  k = tvm::te::reduce_axis(Range(0, n), "k");
146 
147  auto expsum = tvm::te::compute(
148  { m }, [&](Var i) {
149  return tvm::sum(tvm::exp(x(i, k) - max_elem(i)), { k }); });
150 
151  return tvm::te::compute(
152  x->shape, [&](Var i, Var j) {
153  return x(i, j) - max_elem(i) - tvm::log(expsum(i));
154  }, name, tag);
155 }
