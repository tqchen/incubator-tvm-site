44 enum OpPatternKind {
45  // Elementwise operation
46  kElemWise = 0,
47  // Broadcasting operator, can always map output axis to the input in order.
48  // for example :code:`out[i, ax1, j, ax2] = input[i, j]`.
49  // Note that the axis need to be in order so transpose is not a bcast operator.
50  kBroadcast = 1,
51  // Injective operator, can always injectively map output axis to a single input axis.
52  // All injective operator can still be safely fused to injective and reduction.
53  kInjective = 2,
54  // Communicative reduction operator.
55  kCommReduce = 3,
56  // Complex operation, can still fuse elemwise operations into its output.
57  // but cannot chain another complex op
58  kOutEWiseFusable = 4,
59  // The pattern for tuple nodes. Can fuse into subsequent injective ops,
60  // but treated specially
61  kTuple = 7,
62  // Opaque operation, cannot fuse anything.
63  kOpaque = 8
64 };
