552  template<typename ...Args>

554  Array<Stmt> seq;

564  : seq_(seq) {}
565 
566  void operator()(size_t i, const Stmt& stmt) const {
567  if (!stmt.defined()) return;
568  if (auto* op = stmt.as<SeqStmtNode>()) {
569  operator()(0, op->seq);
570  } else if (auto* op = stmt.as<ProducerConsumerNode>()) {
571  // NOTE: The consumer block annotation was not as useful and can be safely dropped.
572  if (!op->is_producer) {
573  operator()(0, op->body);
574  } else {
575  seq_->push_back(stmt);
576  }
577  } else {
578  seq_->push_back(stmt);
579  }
580  }

584  for (auto v : seq) {
