68 inline Tensor dilate(const Tensor& x,
69  Array<PrimExpr> strides,
70  std::string name = "tensor",
71  std::string tag = kInjective) {
72  auto n = x->shape.size();
73  CHECK_EQ(n, strides.size())
74  << "strides size (" << strides.size()
75  << ") must match dimension of x (" << n << ")";
76 
77  Array<PrimExpr> out_shape;
78  for (size_t i = 0; i < n; ++i) {
79  out_shape.push_back(tvm::tir::Simplify(
80  (x->shape[i] - 1) * cast(DataType::Int(32), strides[i] + 1)));
81  }
82 
83  return tvm::te::compute(
84  out_shape,
85  [&](const Array<Var>& indices) {
86  Array<PrimExpr> not_zero;
87  Array<PrimExpr> index_tuple;
88  for (size_t i = 0; i < n; ++i) {
89  if (IsConstInt(strides[i]) && GetConstInt(strides[i]) == 1) {
90  index_tuple.push_back(indices[i]);
91  } else {
92  index_tuple.push_back(indexdiv(indices[i], strides[i]));
93  not_zero.push_back((indexmod(indices[i], strides[i])) == 0);
94  }
95  }
96  if (not_zero.size() > 0) {
97  auto all_not_zero = all(not_zero);
98  return tvm::if_then_else(
99  all_not_zero, x(index_tuple), make_const(x->dtype, 0));
100  }
101  return x(index_tuple);
102  }, name, tag);
103 }
