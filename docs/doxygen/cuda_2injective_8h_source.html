47 inline Schedule schedule_injective_from_existing(Schedule sch, const Tensor& out) {
48  auto fused = detail::Fuse(sch[out], sch[out]->op.as<ComputeOpNode>()->axis);
49  auto target = Target::Current(false);
50  auto num_thread = target->max_num_threads;
