102  traverse = [&](const Operation& op) {
103  // Inline all one-to-one-mapping operators except the last stage (output)
104  if (is_broadcast(op->tag)) {
105  if (!detail::contains(s->outputs, op)) {
106  s[op].compute_inline();
107  }
108  for (auto tensor : op->InputTensors()) {

121  };
