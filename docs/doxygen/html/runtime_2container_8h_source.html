108 template <typename ArrayType, typename ElemType>
109 class InplaceArrayBase {
110  public:
116  const ElemType& operator[](size_t idx) const {
117  size_t size = Self()->GetSize();
118  CHECK_LT(idx, size) << "Index " << idx << " out of bounds " << size << "\n";
119  return *(reinterpret_cast<ElemType*>(AddressOf(idx)));
120  }
121 
127  ElemType& operator[](size_t idx) {
128  size_t size = Self()->GetSize();
129  CHECK_LT(idx, size) << "Index " << idx << " out of bounds " << size << "\n";
130  return *(reinterpret_cast<ElemType*>(AddressOf(idx)));
131  }
132 
136  ~InplaceArrayBase() {
137  if (!(std::is_standard_layout<ElemType>::value &&
138  std::is_trivial<ElemType>::value)) {
139  size_t size = Self()->GetSize();
140  for (size_t i = 0; i < size; ++i) {
141  ElemType* fp = reinterpret_cast<ElemType*>(AddressOf(i));
142  fp->ElemType::~ElemType();
143  }
144  }
145  }
146 
147  protected:
158  template <typename... Args>
159  void EmplaceInit(size_t idx, Args&&... args) {
160  void* field_ptr = AddressOf(idx);
161  new (field_ptr) ElemType(std::forward<Args>(args)...);
162  }
163 
164  private:
170  inline ArrayType* Self() const {
171  return static_cast<ArrayType*>(const_cast<InplaceArrayBase*>(this));
172  }
173 
180  void* AddressOf(size_t idx) const {
181  static_assert(alignof(ArrayType) % alignof(ElemType) == 0 &&
182  sizeof(ArrayType) % alignof(ElemType) == 0,
183  "The size and alignment of ArrayType should respect "
184  "ElemType's alignment.");
185 
186  size_t kDataStart = sizeof(ArrayType);
187  ArrayType* self = Self();
188  char* data_start = reinterpret_cast<char*>(self) + kDataStart;
189  return data_start + idx * sizeof(ElemType);
190  }
191 };

568 inline int String::memncmp(const char* lhs, const char* rhs, size_t lhs_count,
569  size_t rhs_count) {
570  if (lhs == rhs && lhs_count == rhs_count) return 0;
571 
572  for (size_t i = 0; i < lhs_count && i < rhs_count; ++i) {
573  if (lhs[i] < rhs[i]) return -1;
574  if (lhs[i] > rhs[i]) return 1;
575  }
576  if (lhs_count < rhs_count) {
577  return -1;
578  } else if (lhs_count > rhs_count) {
579  return 1;
580  } else {
581  return 0;
582  }
583 }
