50 inline tvm::te::Tensor binarize_pack(const tvm::te::Tensor& data,
51  int axis,
52  std::string name = "PackedInput",
53  std::string tag = "binarize_pack") {
54  auto ishape = data->shape;
55  CHECK_EQ(GetConstInt(ishape[axis]) % 32, 0)
56  << "binarize_pack: axis size must be a multiple of 32";
57 
58  auto n = ishape.size();
59  Array<PrimExpr> oshape;
60  for (size_t i = 0; i < n; ++i) {
61  oshape.push_back(i == static_cast<size_t>(axis) ?
62  tvm::tir::Simplify(indexdiv(ishape[i], 32)) :
63  ishape[i]);
64  }
65 
66  return tvm::te::compute(
67  oshape,
68  [&](const Array<Var>& indices) {
69  Array<PrimExpr> start_idx;
70  for (size_t i = 0; i < n; ++i) {
71  start_idx.push_back(i == static_cast<size_t>(axis) ?
72  indices[i] * 32 :
73  static_cast<PrimExpr>(indices[i]));
74  }
75  auto packed = make_const(DataType::UInt(32), 0);
76  for (size_t j = 0; j < 32; ++j) {
77  Array<PrimExpr> idx;
78  for (size_t i = 0; i < n; ++i) {
79  idx.push_back(i == static_cast<size_t>(axis) ?
80  start_idx[i] + static_cast<int>(j) :
81  start_idx[i]);
82  }
83  auto sign = tvm::cast(DataType::UInt(32), data(idx) >= 0);
84  packed = (packed | sign);
85  if (j == 31) {
86  return packed;
87  }
88  packed = packed << 1;
89  }
90  return packed; // never reached, but suppress compiler warning
91  }, name, tag);
92 }

102 inline tvm::te::Tensor binary_dense(const tvm::te::Tensor& data,
103  const tvm::te::Tensor& weight) {
104  CHECK_EQ(data->shape.size(), 2) << "binary_dense requires 2-D data";
105  CHECK_EQ(weight->shape.size(), 2) << "binary_dense requires 2-D weight";
106  CHECK_EQ(data->dtype, DataType::UInt(32)) << "binary_dense requires uint32 data";
107  CHECK_EQ(weight->dtype, DataType::UInt(32)) << "binary_dense requires uint32 weight";
108 
109  auto batch = data->shape[0];
110  auto in_dim = data->shape[1];
111  auto out_dim = weight->shape[0];
112 
113  auto k = tvm::te::reduce_axis(Range(0, in_dim), "k");
114  auto matmul = tvm::te::compute(
115  { batch, out_dim },
116  [&](Var i, Var j) {
117  return tvm::sum(popcount(data(i, k) ^ weight(j, k)), { k });
118  }, "tensor", "binary_dense");
119 
120  return tvm::te::compute(
121  { batch, out_dim },
122  [&](Var i, Var j) {
123  return 32 * in_dim - 2.0f * matmul(i, j);
124  }, "tensor", kElementWise);
125 }
