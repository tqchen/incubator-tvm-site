63 inline std::vector<int> GetRealAxis(int ndim, const Array<Integer>& axis) {
64  std::vector<int> real_axis;
65  if (!axis.defined() || axis.size() == 0) {
66  for (int i = 0; i < ndim; ++i) {
67  real_axis.push_back(i);
68  }
69  } else {
70  // Use a set so duplicates are removed and the dims are sorted
71  for (auto elem : axis) {
72  int64_t val = elem->value;
73  if (val < 0) {
74  val += ndim;
75  }
76  CHECK_LE(val, ndim) << " exceeds the maximum dimension " << ndim;
77  CHECK_GE(val, 0);
78  real_axis.push_back(static_cast<int>(val));
79  }
80  std::sort(real_axis.begin(), real_axis.end());
81  real_axis.resize(
82  std::unique(real_axis.begin(), real_axis.end()) - real_axis.begin());
83  }
84  return real_axis;
85 }

99 inline Array<PrimExpr> MakeReduceTargetShape(const std::vector<int>& real_axis,
100  const Tensor& data,
101  bool keepdims,
102  bool atleast1d) {
103  auto ndim = data->shape.size();
104  Array<PrimExpr> target_shape;
105  if (keepdims) {
106  for (size_t i = 0; i < ndim; ++i) {
107  if (std::find(real_axis.begin(), real_axis.end(), i) != real_axis.end()) {
108  // real_axis contains i
109  target_shape.push_back(1);
110  } else {
111  target_shape.push_back(data->shape[i]);
112  }
113  }
114  } else {
115  for (size_t i = 0; i < ndim; ++i) {
116  if (std::find(real_axis.begin(), real_axis.end(), i) == real_axis.end()) {
117  // real_axis does not contain i
118  target_shape.push_back(data->shape[i]);
119  }
120  }
121  }
122  if (target_shape.size() == 0 && atleast1d) {
123  target_shape.push_back(1);
124  }
125  return target_shape;
126 }

140 inline Tensor DoCommReduce(const Tensor& data,
141  FReduce func,
142  const Array<PrimExpr>& target_shape,
143  const std::vector<int>& reduce_axes,
144  const std::vector<int>& squeeze_axes) {
145  auto r_axes = MakeReduceAxes(reduce_axes, data);
146  auto compute = [&](const Array<Var>& indices) {
147  Array<PrimExpr> eval_range;
148  Array<Var> eval_indices;
149  int arg_counter = 0;
150  int red_counter = 0;
151 
152  for (size_t i = 0; i < data->shape.size(); ++i) {
153  bool squeeze_i = std::find(squeeze_axes.begin(), squeeze_axes.end(), i) != squeeze_axes.end();
154  if (std::find(reduce_axes.begin(), reduce_axes.end(), i) != reduce_axes.end()) {
155  // real_axis contains i
156  eval_range.push_back(r_axes[red_counter]);
157  eval_indices.push_back(r_axes[red_counter]->var);
158  red_counter++;
159  arg_counter += !squeeze_i;
160  continue;
161  }
162  eval_range.push_back(indices[arg_counter]);
163  arg_counter++;
164  }
165 
166  return func(data(eval_range), r_axes);
167  };
168 
169  return tvm::te::compute(target_shape, compute, data->op->name + "_red", kCommReduce);
170 }

211 inline Tensor CommReduceIdx(const Tensor& data,
212  const Array<Integer>& axis,
213  FCommReduce func,
214  bool keepdims,
215  bool atleast1d) {
216  auto ndim = data->shape.size();
217  CHECK_NE(ndim, 0) << "Cannot reduce a 0 dim Tensor";
218  auto real_axis = GetRealAxis(static_cast<int>(ndim), axis);
219  auto reduce_axes = MakeReduceAxes(real_axis, data);
220  auto target_shape = MakeReduceTargetShape(real_axis, data, keepdims, atleast1d);
221 
222  auto compute = [ndim, keepdims, &real_axis, &reduce_axes, &func, &data]
223  (const Array<Var>& indices) {
224  Array<PrimExpr> eval_range;
225  Array<PrimExpr> eval_indices;
226  int arg_counter = 0;
227  int red_counter = 0;
228 
229  for (size_t i = 0; i < ndim; ++i) {
230  if (std::find(real_axis.begin(), real_axis.end(), i) != real_axis.end()) {
231  // real_axis contains i
232  eval_range.push_back(reduce_axes[red_counter]);
233  eval_indices.push_back(reduce_axes[red_counter]->var);
234  red_counter++;
235  } else {
236  if (!keepdims) {
237  eval_range.push_back(indices[arg_counter]);
238  arg_counter++;
239  } else {
240  eval_range.push_back(indices[i]);
241  }
242  }
243  }
244 
245  Array<PrimExpr> ravel_shape;
246  for (auto i : real_axis) {
247  ravel_shape.push_back(data->shape[i]);
248  }
249  auto idx = detail::RavelIndex(eval_indices, ravel_shape);
250  return func({ idx, data(eval_range) }, reduce_axes, nullptr);
251  };
252 
253  auto temp_idx_val = tvm::te::compute(target_shape, compute,
254  data->op->name + "_red_temp", kCommReduceIdx);
255  auto temp_idx = temp_idx_val[0];
256  auto temp_val = temp_idx_val[1];
257  return tvm::te::compute(
258  target_shape,
259  [&temp_idx](const Array<Var>& indices) { return temp_idx(indices); },
260  data->op->name + "_red",
261  kCommReduceIdx);
262 }

279 inline FCommReduce MakeCommReducer(FCombine fcombine,
280  FIdentity fidentity,
281  std::string name = "reduce") {
282  return [fcombine, fidentity, name]
283  (Array<PrimExpr> exprs, const Array<IterVar>& axis, PrimExpr* condition) {
284  Array<Var> lhs, rhs;
285  std::vector<DataType> dtypes;
286 
287  for (size_t i = 0; i < exprs.size(); ++i) {
288  auto dtype = exprs[i].dtype();
289  dtypes.push_back(dtype);
290  lhs.push_back(var(name + "_lhs_" + std::to_string(i), dtype));
291  rhs.push_back(var(name + "_rhs_" + std::to_string(i), dtype));
292  }
293 
294  auto result = fcombine(lhs, rhs);
295  auto id_elem = fidentity(dtypes);
296  auto cond = condition != nullptr ? *condition : tir::const_true();
297 
298  auto combiner = tvm::tir::CommReducerNode::make(lhs, rhs, result, id_elem);
299  Array<PrimExpr> outputs;
300  for (size_t i = 0; i < exprs.size(); ++i) {
301  outputs.push_back(
302  tvm::tir::ReduceNode::make(combiner, exprs, axis, cond, static_cast<int>(i)));
303  }
304  return outputs;
305  };
306 }

343 inline Tensor collapse_sum(const Tensor& data, Array<PrimExpr> target_shape) {
344  CHECK_GE(data->shape.size(), target_shape.size());
345  auto ishape = detail::GetConstIntValues(data->shape, "ishape");
346  auto oshape = detail::GetConstIntValues(target_shape, "oshape");
347 
348  std::vector<int> reduce_axes;
349  std::vector<int> squeeze_axes;
350  for (int i_ax = ishape.size() - 1,
351  o_ax = oshape.size() - 1; i_ax >= 0; --i_ax) {
352  if (o_ax >= 0 && ishape[i_ax] == oshape[o_ax]) {
353  --o_ax;
354  continue;
355  }
356  reduce_axes.push_back(i_ax);
357  if (o_ax < 0) { // squeeze o_ax if was added during expansion
358  squeeze_axes.push_back(i_ax);
359  } else if (oshape[o_ax] == 1) {
360  --o_ax;
361  }
362  }
363 
364  if (reduce_axes.size() == 0) return topi::identity(data, "tensor", kCommReduce);
365 
366  std::reverse(reduce_axes.begin(), reduce_axes.end());
367  std::reverse(squeeze_axes.begin(), squeeze_axes.end());
368  return DoCommReduce(data, tvm::sum, target_shape, reduce_axes, squeeze_axes);
369 }

469 inline Tensor argmin(const Tensor& data,
470  const Array<Integer>& axis,
471  bool keepdims = false,
472  bool atleast1d = false) {
473  auto fcombine = [](Array<Var> lhs, Array<Var> rhs) {
474  Array<PrimExpr> result;
475  result.push_back(tvm::tir::SelectNode::make(lhs[1] <= rhs[1], lhs[0], rhs[0])); // idx
476  result.push_back(tvm::tir::SelectNode::make(lhs[1] <= rhs[1], lhs[1], rhs[1])); // val
477  return result;
478  };
479  auto fidentity = [](std::vector<DataType> types) {
480  Array<PrimExpr> result;
481  result.push_back(tvm::tir::make_const(types[0], -1)); // idx
482  result.push_back(tvm::max_value(types[1])); // val
483  return result;
484  };
485  auto func = MakeCommReducer(fcombine, fidentity, "argmin");
486  return CommReduceIdx(data, axis, func, keepdims, atleast1d);
