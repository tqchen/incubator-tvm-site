170 class TVM_DLL StmtMutator :
171  protected StmtFunctor<Stmt(const Stmt&)> {
172  public:
181  Stmt operator()(Stmt stmt) {
182  allow_copy_on_write_ = true;
183  return VisitStmt(stmt);
184  }
185 
186  protected:
187  // We perform copy on write optimizations on the StmtMutator
188  // so that an unique copy of parent can be mutated inplace
189  // when some of its children changed.
190  // We only do such optimization for Stmt nests(instead of Exprs) for now
191  // as Stmt's parent state is more likely remain unchanged when one of
192  // its child block changes.
197  bool allow_copy_on_write_{false};
207  template<typename TNode>
208  ObjectPtr<TNode> CopyOnWrite(const TNode* node) {
209  if (allow_copy_on_write_) {
210  // return the old node.
211  return runtime::GetObjectPtr<TNode>(const_cast<TNode*>(node));
212  } else {
213  // Make a new copy of the node.
214  // need to rely on the default copy constructor
215  return runtime::make_object<TNode>(*node);
216  }
217  }
224  Stmt VisitStmt(const Stmt& stmt) override {
225  if (allow_copy_on_write_ && !stmt.unique()) {
226  allow_copy_on_write_ = false;
227  Stmt ret = StmtFunctor::VisitStmt(stmt);
228  allow_copy_on_write_ = true;
229  return ret;
230  } else {
231  return StmtFunctor::VisitStmt(stmt);
232  }
233  }
241  virtual PrimExpr VisitExpr(const PrimExpr& e) {
242  return e;
243  }
