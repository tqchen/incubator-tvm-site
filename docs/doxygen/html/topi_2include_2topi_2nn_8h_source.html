175 inline tvm::te::Tensor pad(const tvm::te::Tensor& t,
176  const tvm::Array<tvm::PrimExpr>& pad_before,
177  tvm::Array<tvm::PrimExpr> pad_after = tvm::Array<tvm::PrimExpr>(),
178  PrimExpr pad_value = PrimExpr(),
179  std::string name = "T_pad",
180  std::string tag = kElementWise,
181  std::string pad_mode = "constant") {
182  if (pad_after.size() < pad_before.size()) {
183  for (size_t i = pad_after.size(); i < pad_before.size(); ++i) {
184  pad_after.push_back(pad_before[i]);
185  }
186  }
187  CHECK_GE(pad_before.size(), 1);
188  CHECK_EQ(pad_before.size(), pad_after.size());
189  tvm::Array<tvm::PrimExpr> output_shape;
190  tvm::Array<tvm::PrimExpr> pad_before_int32;
191  tvm::Array<tvm::PrimExpr> pad_after_int32;
192  for (const auto &ele : pad_before) {
193  pad_before_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
194  }
195  for (const auto &ele : pad_after) {
196  pad_after_int32.push_back(tvm::cast(tvm::DataType::Int(32), ele));
197  }
198  for (size_t i = 0; i < t->shape.size(); ++i) {
199  if (i >= pad_before.size()) {
200  output_shape.push_back(t->shape[i]);
201  } else {
202  output_shape.push_back(
203  tvm::tir::Simplify(t->shape[i] + pad_before_int32[i] + pad_after_int32[i]));
204  }
205  }
206 
207  if (!pad_value.defined()) {
208  pad_value = tvm::tir::make_const(t->dtype, 0);
209  }
210  auto l = [&](tvm::Array<tvm::tir::Var> ovars) {
211  tvm::Array<tvm::PrimExpr> indices;
212  tvm::Array<tvm::PrimExpr> sel;
213  tvm::Array<tvm::PrimExpr> pad_idx;
214  for (size_t i = 0; i < t->shape.size(); ++i) {
215  if (i >= pad_before_int32.size()) {
216  indices.push_back(ovars[i]);
217  continue;
218  }
219  if (!topi::detail::EqualCheck(pad_before_int32[i], 0)) {
220  sel.push_back(ovars[i] >= pad_before_int32[i]);
221  indices.push_back(ovars[i] - pad_before_int32[i]);
222  } else {
223  indices.push_back(ovars[i]);
224  }
225  if (!topi::detail::EqualCheck(pad_after_int32[i], 0)) {
226  sel.push_back(tvm::tir::Simplify(ovars[i] < pad_before_int32[i] + t->shape[i]));
227  }
228  if (pad_mode == "edge") {
229  pad_idx.push_back(tvm::if_then_else(
230  ovars[i] < pad_before[i],
231  0,
232  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
233  t->shape[i] - 1,
234  ovars[i] - pad_before[i])));
235  } else if (pad_mode == "reflect") {
236  pad_idx.push_back(tvm::if_then_else(
237  ovars[i] < pad_before[i],
238  pad_before[i] - ovars[i],
239  tvm::if_then_else(ovars[i] >= pad_before[i] + t->shape[i],
240  t->shape[i] * 2 - ovars[i] + pad_before[i] - 2,
241  ovars[i] - pad_before[i])));
242  }
243  }
244  if (sel.size() != 0) {
245  if (pad_mode == "constant") {
246  return tvm::if_then_else(
247  detail::Map(sel, tvm::tir::AndNode::make), t(indices), pad_value);
248  } else if (pad_mode == "edge" || pad_mode == "reflect") {
249  return tvm::if_then_else(
250  detail::Map(sel, tvm::tir::AndNode::make), t(indices), t(pad_idx));
251  }
252  }
253  return t(indices);
254  };
255  return tvm::te::compute(output_shape, l, name, tag);
256 }

278 inline tvm::te::Tensor conv2d_nchw(const tvm::te::Tensor& I,
279  const tvm::te::Tensor& W,
280  int pad_h = 0,
281  int pad_w = 0,
282  int stride_h = 1,
283  int stride_w = 1,
284  std::string name = "T_conv2d_nchw",
285  std::string tag = kConv2dNCHW) {
286  CHECK_EQ(4, I->shape.size());
287  CHECK_EQ(4, W->shape.size());
288  auto pH = I->shape[2];
289  auto pW = I->shape[3];
290  tvm::Array<tvm::PrimExpr> output_shape{
291  I->shape[0], // B
292  W->shape[0], // O
293  indexdiv(I->shape[2] - W->shape[2] + 2 * pad_h, stride_h) + 1, // H
294  indexdiv(I->shape[3] - W->shape[3] + 2 * pad_w, stride_w) + 1 // W
295  };
296  auto i = tvm::te::reduce_axis(tvm::Range{0, I->shape[1]}, "i");
297  auto kh = tvm::te::reduce_axis(tvm::Range{0, W->shape[2]}, "kh");
298  auto kw = tvm::te::reduce_axis(tvm::Range{0, W->shape[3]}, "kw");
299  auto T = (pad_h == 0 && pad_w == 0)
300  ? I
301  : pad(I, {tvm::PrimExpr(0), tvm::PrimExpr(0), pad_h, pad_w});
302  auto l = [&](tvm::tir::Var b, tvm::tir::Var o, tvm::tir::Var h, tvm::tir::Var w) {
303  return tvm::sum(
304  T(b, i, stride_h * h + kh, stride_w * w + kw) * W(o, i, kh, kw),
305  {i, kh, kw});
306  };
307  return tvm::te::compute(output_shape, l, name, tag);
308 }
