44 inline BroadcastHelper BroadcastShape(const tvm::Array<tvm::PrimExpr>& shape1,
45  const tvm::Array<tvm::PrimExpr>& shape2) {
46  BroadcastHelper bh;
47  int s1_size = shape1.size();
48  int s2_size = shape2.size();
49  tvm::PrimExpr one(1);
50  int i;
51  for (i = 1; i <= std::min(s1_size, s2_size); ++i) {
52  // TODO(@icemelon9): Need to revisit this part
53  const VarNode* var1 = shape1[s1_size - i].as<VarNode>();
54  const VarNode* var2 = shape2[s2_size - i].as<VarNode>();
55  bh.all_vars.push_front(tvm::tir::Var());
56  if (topi::detail::EqualCheck(shape1[s1_size - i], shape2[s2_size - i])) {
57  bh.common_shape.push_front(shape1[s1_size - i]);
58  bh.vars1.push_front(bh.all_vars[0]);
59  bh.vars2.push_front(bh.all_vars[0]);
60  } else if (topi::detail::EqualCheck(one, shape1[s1_size - i])) {
61  CHECK(!topi::detail::EqualCheck(one, shape2[s2_size - i]));
62  bh.common_shape.push_front(shape2[s2_size - i]);
63  bh.vars2.push_front(bh.all_vars[0]);
64  } else if (topi::detail::EqualCheck(one, shape2[s2_size - i])) {
65  bh.common_shape.push_front(shape1[s1_size - i]);
66  bh.vars1.push_front(bh.all_vars[0]);
67  } else if (var1 && var2) {
68  bh.common_shape.push_front(max(shape1[s1_size - i], shape2[s2_size - i]));
69  bh.vars1.push_front(bh.all_vars[0]);
70  bh.vars2.push_front(bh.all_vars[0]);
71  } else if (var1) {
72  bh.common_shape.push_front(shape2[s2_size - i]);
73  bh.vars2.push_front(bh.all_vars[0]);
74  bh.vars1.push_front(bh.all_vars[0]);
75  } else if (var2) {
76  bh.common_shape.push_front(shape1[s1_size - i]);
77  bh.vars1.push_front(bh.all_vars[0]);
78  bh.vars2.push_front(bh.all_vars[0]);
79  } else {
80  CHECK(false) << "Incompatible broadcast dims: " << shape1[s1_size - i]
81  << " and " << shape2[s2_size - i] << " in: "
82  << tvm::Array<tvm::PrimExpr>(shape1.begin(), shape1.end())
83  << " and "
84  << tvm::Array<tvm::PrimExpr>(shape2.begin(), shape2.end());
85  }
86  }
87  // Remaining dimensions whether on shape1 or shape2 can always be completed
88  auto max_size = std::max(s1_size, s2_size);
89  auto& shape = (s1_size > s2_size) ? shape1 : shape2;
90  auto& vars = (s1_size > s2_size) ? bh.vars1 : bh.vars2;
91  for (; i <= max_size; ++i) {
92  bh.all_vars.push_front(tvm::tir::Var());
93  bh.common_shape.push_front(shape[max_size - i]);
94  vars.push_front(bh.all_vars[0]);
95  }
96  return bh;
97 }

99 inline tvm::Array<tvm::PrimExpr> InputIndexFromBroadcast(
100  const tvm::Array<tvm::tir::Var>& ovars,
101  const tvm::te::Tensor& T,
102  const std::deque<tvm::tir::Var>& my_vars,
103  const std::deque<tvm::tir::Var>& all_vars) {
104  tvm::Array<tvm::PrimExpr> ivars;
105  CHECK_EQ(ovars.size(), all_vars.size());
106  // N^2, could use a map but NBD.
107  size_t expected_dims = T->shape.size();
108  for (size_t i = 0; i < ovars.size(); ++i) {
109  bool found = false;
110  for (size_t j = 0; j < my_vars.size(); ++j) {
111  if (all_vars[i].same_as(my_vars[j])) {
112  ivars.push_back(ovars[i]);
113  found = true;
114  break;
115  }
116  }
117  // Only inject 0 here if we have not yet reached the dimension of I
118  // (i.e. this must be a 1)
119  if (!found && (ovars.size() - i) <= expected_dims) {
120  ivars.push_back(tvm::tir::make_zero(ovars[i].dtype()));
121  }
122  }
123  CHECK(expected_dims == ivars.size());
124  return ivars;
125 }
