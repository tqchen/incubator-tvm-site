52  int size,

57  std::string name = "tensor",
58  std::string tag = kBroadcast) {
59  CHECK_EQ(data->shape.size(), 4) << "LRN requires 4-D input";
60  CHECK_EQ(size % 2, 1) << "size should be odd number";
61  CHECK(axis == 1 || axis == 3) << "axis should be 1 or 3 for NCHW and NHWC";
62  auto input_shape = data->shape;
63  Array<PrimExpr> pad_before{ 0, 0, 0, 0};
64  Array<PrimExpr> pad_after{ 0, 0, 0, 0};
65  pad_before.Set(axis, static_cast<PrimExpr>(size/2));
66  pad_after.Set(axis, static_cast<PrimExpr>(size/2));
67  auto pad_data = pad(data, pad_before, pad_after, 0, "pad_data");
68  auto rxs = tvm::te::reduce_axis(Range(0, size), "rxs");
69  Tensor sqr_sum;
70  if (axis == 1) {
71  sqr_sum = tvm::te::compute(input_shape,
72  [&](Var i, Var l, Var j, Var k) {
73  return tvm::sum(pad_data(i, l + rxs, j, k) *
74  pad_data(i, l + rxs, j, k),
75  {rxs});
76  });
77  } else if (axis == 3) {
78  sqr_sum = tvm::te::compute(input_shape,
79  [&](Var i, Var l, Var j, Var k) {
80  return tvm::sum(pad_data(i, l, j, k + rxs) *
81  pad_data(i, l, j, k + rxs),
82  {rxs});
83  });
84  }
85  auto sqrt_sum_up = tvm::te::compute(
86  input_shape,
87  [&](Var i, Var j, Var k, Var l) {
88  return tvm::pow(bias +
89  (div(alpha * sqr_sum(i, j, k, l), size)),
90  beta);
91  });
92  return topi::divide(data, sqrt_sum_up);
