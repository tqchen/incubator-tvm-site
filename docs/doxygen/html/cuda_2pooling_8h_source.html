48 inline Schedule schedule_pool(const Target &target, const Array<Tensor>& outs) {
49  Array<Operation> out_ops;
50  for (auto t : outs) {
51  out_ops.push_back(t->op);
52  }
53  auto s = create_schedule(out_ops);

55  auto _schedule = [&](const Tensor& padded_input, const Tensor& pool) {
56  if (padded_input->op->IsInstance<ComputeOpNode>()) {
57  s[padded_input].compute_inline();
58  }
59  auto num_thread = target->max_num_threads;
60  Tensor out;
61  Tensor OL;
62  if (detail::contains(s->outputs, pool->op)) {
63  out = pool;
64  OL = s.cache_write(pool, "local");
65  } else {
66  out = outs[0]->op.output(0);
67  s[pool].set_scope("local");
68  }
69  auto fused = detail::Fuse(s[out], s[out]->op.as<ComputeOpNode>()->axis);
70  IterVar bx, tx;
71  s[out].split(fused, num_thread, &bx, &tx);
72  s[out].bind(bx, tvm::te::thread_axis(Range(), "blockIdx.x"));
73  s[out].bind(tx, tvm::te::thread_axis(Range(), "threadIdx.x"));
74  if (detail::contains(s->outputs, pool->op)) {
75  s[OL].compute_at(s[out], tx);
76  } else {
77  s[pool].compute_at(s[out], tx);
78  }
79  };

81  std::function<void(Operation)> traverse;
82  traverse = [&](const Operation& op) {
83  // Inline all one-to-one-mapping operators except the last stage (output)
84  if (is_broadcast(op->tag)) {
85  if (!detail::contains(s->outputs, op)) {
86  s[op].compute_inline();
87  }
88  for (auto tensor : op->InputTensors()) {
89  if (tensor->op->InputTensors().size() > 0) {
90  traverse(tensor->op);
91  }
92  }
93  } else if (op->tag.rfind("pool", 0) == 0) {
94  // If tag starts with pool
95  auto padded_input = op->InputTensors()[0];
96  auto pool = op.output(0);
97  _schedule(padded_input, pool);
98  } else {
99  LOG(ERROR) << "Unsupported operator " << op->tag;
100  }
101  };
102 
103  traverse(outs[0]->op);
104  return s;
105 }

122  auto _schedule = [&](const Tensor& pool) {
123  auto num_thread = 8;
124  auto block_x = tvm::te::thread_axis(Range(), "blockIdx.x");
125  auto block_y = tvm::te::thread_axis(Range(), "blockIdx.y");
126  auto thread_x = tvm::te::thread_axis(Range(0, num_thread), "threadIdx.x");
127  auto thread_y = tvm::te::thread_axis(Range(0, num_thread), "threadIdx.y");
128  Tensor out;
129  Tensor OL;
130  if (detail::contains(s->outputs, pool->op)) {
131  out = pool;
132  OL = s.cache_write(pool, "local");
133  } else {
134  out = outs[0]->op.output(0);
135  s[pool].set_scope("local");
136  }
137 
138  auto i = s[out]->op.as<ComputeOpNode>()->axis[0];
139  auto c = s[out]->op.as<ComputeOpNode>()->axis[1];
140 
141  IterVar by, ty;
142  s[out].split(i, num_thread, &by, &ty);
143  IterVar bx, tx;
144  s[out].split(c, num_thread, &bx, &tx);

152  s[OL].compute_at(s[out], tx);
153  } else {
154  s[pool].compute_at(s[out], tx);
