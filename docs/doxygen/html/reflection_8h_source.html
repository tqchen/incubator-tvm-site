364  uint32_t tindex = T::RuntimeTypeIndex();
365  if (tindex >= fvisit_attrs_.size()) {
366  fvisit_attrs_.resize(tindex + 1, nullptr);
367  fcreate_.resize(tindex + 1, nullptr);
368  fglobal_key_.resize(tindex + 1, nullptr);
369  fsequal_reduce_.resize(tindex + 1, nullptr);
370  fshash_reduce_.resize(tindex + 1, nullptr);
371  }
372  // functor that implemnts the redirection.
373  fvisit_attrs_[tindex] =
374  ::tvm::detail::SelectVisitAttrs<T, TraitName>::VisitAttrs;
375 
376  fsequal_reduce_[tindex] =
377  ::tvm::detail::SelectSEqualReduce<T, TraitName>::SEqualReduce;
378 
379  fshash_reduce_[tindex] =
380  ::tvm::detail::SelectSHashReduce<T, TraitName>::SHashReduce;
381 
382  return Registry(this, tindex);
