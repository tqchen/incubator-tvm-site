66  const PrimExpr max_y, const PrimExpr max_x) {

72  auto y1 = tvm::if_then_else((yc > max_y), max_y, yc);
73  auto y_lerp = in_y - yf;

75  auto in_x = indices[3];

80  auto x1 = tvm::if_then_else((xc > max_x), max_x, xc);
81  auto x_lerp = in_x - xf;

84  auto B = input(indices[0], indices[1], y0, x1);
85  auto C = input(indices[0], indices[1], y1, x0);
